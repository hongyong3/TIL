```python
T = int(input())
for test_case in range(T):
    N = int(input())

    idx = 0
    arr1 = [0]
    arr2 = [0] * (N + 1)
    maxCount, maxNum = 0, 0

    for i in range(1, N):
        if pow(i, 3) < N:
            arr1.append(pow(i, 3))
            arr2[pow(i, 3)] = 1
        else:
            break
    arr1.append(0)

    for i in range(N + 1):
        if i == arr1[idx]:
            idx += 1
        else:
            arr2[i] = arr2[i - arr1[idx - 1]] + 1

    maxCount = max(arr2)
    maxNum = len(arr2) - 1 - arr2[::-1].index(maxCount)
    print("#{} {} {}".format(test_case + 1, maxCount, maxNum))
```
